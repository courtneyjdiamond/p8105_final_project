Neighborhoods Analyses

  • Rides per start neighborhood per month
neighborhood_month = citibike_zip_neighborhoods |>
  mutate(month = month(start_time)) |>
  group_by(start_uhf34_neighborhood, month) |>
  summarize(n_rides = n())

neighborhood_month |> 
  ggplot(aes(x = month, y = n_rides, color = start_uhf34_neighborhood)) +
  geom_point() +                  # Add points for each observation
  geom_line() +                   # Add lines to connect points
  labs(x = "Month", y = "n_rides") # Label the axes
